def create
  @account = Account.new(account_params)
  @account.generate_token!
  if @account.save
    render 'api/v1/accounts/show', :layout => false
  else
    @message = @account.errors.first[0].to_s+" "+@account.errors.first[1].to_s
    render 'api/v1/error', :status => :unprocessable_entity
  end
end
